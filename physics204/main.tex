	% #2 = after alignment
	&
	\begingroup
	\settowidth\dlf{$\displaystyle #1$}
	\addtolength\dlf{\fboxsep+\fboxrule}
	\hspace{-\dlf}
	\boxed{#1 #2}
	\endgroup
}
